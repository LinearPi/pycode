def drop_stopword(contents, stopwords):
    content_clean = []
    all_words = []
    for line in contents:
        line_clean = []
        for word in line:
            if word in stopwords:
                continue
            line_clean.append(word)            
            all_words.append(str(word))
        content_clean.append(line_clean)
    return content_clean, all_words
